Covid with Streamlit
""" 위 코드를 streamlit을 사용한 코드를 변경해주세요. 1. coivd 디렉터리는 사용자 입력을 받음 2. 입력받은 디렉터리 경로는 현재 스크립트 파일을 기준으로 하위에 있음 """ import pandas as pd import os import glob import streamlit as st # Streamlit 앱의 제목 설정 st.title('2020년 Covid 현황') # 사용자로부터 디렉터리 경로 입력 받기 input_folder = st.text_input('디렉터리 경로 입력:', 'covid') # 스크립트 파일이 위치한 디렉토리를 기준으로 입력받은 경로 조합 base_path = os.path.abspath(os.path.dirname(__file__)) folder_path = os.path.join(base_path, input_folder) # 경로 확인 버튼 if st.button('데이터 처리 및 저장'): if not os.path.exists(folder_path): st.error('입력한 경로가 존재하지 않습니다. 경로를 확인해주세요.') else: # covid 디렉터리에서 모든 csv 파일 목록을 가져옴 csv_files = glob.glob(os.path.join(folder_path, '*.csv')) # 병합된 데이터를 저장할 빈 데이터프레임 생성 merged_df = pd.DataFrame() for file in csv_files: # 파일에서 'Country_Region', 'Confirmed' 컬럼만 읽어옴 df = pd.read_csv(file, usecols=['Country_Region', 'Confirmed']) # 'Country_Region'별로 그룹핑하고 'Confirmed'의 합계를 구함 grouped_df = df.groupby('Country_Region', as_index=False).sum() # 파일 이름에서 확장자를 제거하고 '-'를 '/'로 변경함 column_name = os.path.basename(file).replace('.csv', '').replace('-', '/') # 'Confirmed' 컬럼 이름을 변경함 grouped_df.columns = ['Country_Region', column_name] # 병합된 데이터프레임에 현재 데이터프레임을 병합함 if merged_df.empty: merged_df = grouped_df else: merged_df = pd.merge(merged_df, grouped_df, on='Country_Region', how='outer') # NaN 값을 0으로 채움 merged_df.fillna(0, inplace=True) # 병합된 데이터를 화면에 표시 st.dataframe(merged_df) merged_df.to_excel('2020_Covid_Status.xlsx', index=False)

import streamlit as st import pandas as pd from datetime import datetime as dt import datetime # 버튼 클릭 button = st.button('버튼을 눌러보세요') if button: st.write(':blue[버튼]이 눌렸습니다 :sparkles:') # 파일 다운로드 버튼 # 샘플 데이터 생성 dataframe = pd.DataFrame({ 'first column': [1, 2, 3, 4], 'second column': [10, 20, 30, 40], }) # 다운로드 버튼 연결 st.download_button( label='CSV로 다운로드', data=dataframe.to_csv(), file_name='sample.csv', mime='text/csv' ) # 체크 박스 agree = st.checkbox('동의 하십니까?') if agree: st.write('동의 해주셔서 감사합니다 :100:') # 라디오 선택 버튼 mbti = st.radio( '당신의 MBTI는 무엇입니까?', ('ISTJ', 'ENFP', '선택지 없음')) if mbti == 'ISTJ': st.write('당신은 :blue[현실주의자] 이시네요') elif mbti == 'ENFP': st.write('당신은 :green[활동가] 이시네요') else: st.write("당신에 대해 :red[알고 싶어요]:grey_exclamation:") # 선택 박스 mbti = st.selectbox( '당신의 MBTI는 무엇입니까?', ('ISTJ', 'ENFP', '선택지 없음'), index=2 ) if mbti == 'ISTJ': st.write('당신은 :blue[현실주의자] 이시네요') elif mbti == 'ENFP': st.write('당신은 :green[활동가] 이시네요') else: st.write("당신에 대해 :red[알고 싶어요]:grey_exclamation:") # 다중 선택 박스 options = st.multiselect( '당신이 좋아하는 과일은 뭔가요?', ['망고', '오렌지', '사과', '바나나'], ['망고', '오렌지']) st.write(f'당신의 선택은: :red[{options}] 입니다.') # 슬라이더 values = st.slider( '범위의 값을 다음과 같이 지정할 수 있어요:sparkles:', 0.0, 100.0, (25.0, 75.0)) st.write('선택 범위:', values) start_time = st.slider( "언제 약속을 잡는 것이 좋을까요?", min_value=dt(2020, 1, 1, 0, 0), max_value=dt(2020, 1, 7, 23, 0), value=dt(2020, 1, 3, 12, 0), step=datetime.timedelta(hours=1), format="MM/DD/YY - HH:mm") st.write("선택한 약속 시간:", start_time) # 텍스트 입력 title = st.text_input( label='가고 싶은 여행지가 있나요?', placeholder='여행지를 입력해 주세요' ) st.write(f'당신이 선택한 여행지: :violet[{title}]') # 숫자 입력 number = st.number_input( label='나이를 입력해 주세요.', min_value=10, max_value=100, value=30, step=5 ) st.write('당신이 입력하신 나이는: ', number)

import streamlit as st import random import datetime st.title(':sparkles:로또 생성기:sparkles:') def generate_lotto(): lotto = set() while len(lotto) < 6: number = random.randint(1, 46) lotto.add(number) lotto = list(lotto) lotto.sort() return lotto button = st.button('로또를 생성해 주세요!') if button: for i in range(1, 6): st.subheader(f'{i}. 행운의 번호: :green[{generate_lotto()}]') st.write(f"생성된 시각: {datetime.datetime.now().strftime('%Y-%m-%d %H:%M')}")
