당신은 유능한 파이썬 개발자입니다. PDF 파일로 저장된 물품 공급 계약서에서 원하는 항목만 추출하여 엑셀로 저장하는 프로그램을 구현해주세요 구현할 세부 내용은 다음과 같습니다. # 제목 물품 공급 계약 #조건 1. streamlit을 사용하여 구현함 2. 작업하는 파일의 기본 위치는 현재 실행중인 스크립트 파일이 저장된 위치임 3. 물품 공급 계약서(PDF 파일)가 저장된 디렉터리를 입력 받음 4. 추출 항목들을 체크할수 있는 위젯을 생성. 선택할 수 있는 추출 항목은 다음과 같음 계약 금액, 회사명, 대표자, 연락처, 주소 5. 다음은 선택된 추출 항목을 추출하기 위한 조건임 계약 금액 : "일금"과 "부가세포함" 문자열 사이에 있는 숫자만 추출후 int로 변경해줌 회사명 : "회 사 명:" 다음에 나오는 글자들을 추출한 후 두번째 값을 사용함 대표자 : "대 표 자:" 다음에 나오는 글자들을 추출한 후 두번째 값을 사용하고 "(인)" 글자는 삭제함 연락처 : "연 락 처:" 다음에 나오는 글자들을 추출한 후 두번째 값을 사용함 주소 : "주 소:" 다음에 나오는 글자들을 추출한 후 두번째 값을 사용함 6. 5번에서 선택한 항목들을 모든 PDF 파일에서 추출하여 한개의 DataFrame에 담음 7. 6번의 DataFrame을 화면에 출력함 8. 버튼을 클릭하면 5번의 DataFrame을 엑셀로 저장함

import streamlit as st import pandas as pd import pdfplumber import re import os # 스트림릿 앱 타이틀 설정 st.title('물품 공급 계약서 정보 추출기') # 디렉터리 입력 받기 directory = st.text_input('PDF 파일이 저장된 디렉터리 경로를 입력하세요:', r'C:\test_chatgptRPA\05 PDF 정보 추출하기\data') # 추출할 항목 선택 selected_options = st.multiselect('추출할 항목을 선택하세요:', ['계약 금액', '회사명', '대표자', '연락처', '주소'], default=['계약 금액', '회사명', '대표자', '연락처', '주소']) # 추출 함수 정의 def extract_info(pdf_path): with pdfplumber.open(pdf_path) as pdf: all_text = "" for page in pdf.pages: all_text += page.extract_text() or "" data = {} for option in selected_options: if option == '계약 금액': result = re.search(r"일금(.+?)부가세포함", all_text) if result: money = int(re.sub(r'[^\d]', '', result.group(1))) # 숫자만 추출 후 정수 변환 data['계약 금액'] = money elif option == '회사명': result = re.findall(r"회\s*사\s*명:\s*(.+)", all_text) if result and len(result) >= 2: data[option] = result[1].strip() elif option == '대표자': result = re.findall(r"대\s*표\s*자:\s*(.+?)\s*\(인\)", all_text) if result and len(result) >= 2: data[option] = result[1].replace("(인)", "").strip() elif option == '연락처': result = re.findall(r"연\s*락\s*처:\s*(.+)", all_text) if result and len(result) >= 2: data[option] = result[1].strip() elif option == '주소': result = re.findall(r"주\s*소:\s*(.+)", all_text) if result and len(result) >= 2: data[option] = result[1].strip() return data # DataFrame 초기화 df = pd.DataFrame() # 데이터 추출 및 DataFrame 생성 if st.button('PDF 파일 처리'): new_data = [] for filename in os.listdir(directory): if filename.endswith('.pdf'): pdf_path = os.path.join(directory, filename) extracted_data = extract_info(pdf_path) new_data.append(extracted_data) if new_data: df = pd.DataFrame(new_data) st.session_state['data'] = df # 다른 버튼을 클릭하면 다시 초기화되기 때문에 저장함 st.write(df) # DataFrame을 엑셀로 저장 if st.button('엑셀로 저장'): output_file = os.path.join(directory, '추출된_데이터.xlsx') save_data = st.session_state['data'] # 저장된 데이터 사용함 save_data.to_excel(output_file, index=False) st.success(f'데이터가 성공적으로 저장되었습니다: {output_file}')
