################# ìœ íŠœë¸Œ ë‚´ìš© ìš”ì•½í•˜ê¸° ################### import os from langchain_community.document_loaders import YoutubeLoader from langchain.text_splitter import RecursiveCharacterTextSplitter from langchain_openai import ChatOpenAI from langchain_core.prompts import PromptTemplate from langchain.chains.summarize import load_summarize_chain URL = "https://www.youtube.com/watch?v=Pn-W41hC764" loader = YoutubeLoader.from_youtube_url(URL) transcript = loader.load() text_splitter = RecursiveCharacterTextSplitter(chunk_size=4000, chunk_overlap=0) text = text_splitter.split_documents(transcript) api_key = os.getenv('OPENAI_API_KEY') llm = ChatOpenAI(temperature=0, openai_api_key=api_key, max_tokens=3000, model="gpt-4o-mini", request_timeout=120) # ê°ê°ì˜ chunck ë¥¼ ìš”ì•½í•˜ê¸° prompt = PromptTemplate( template="""Summarize the youtube video whose transcript is provided within backticks ```{text}``` """, input_variables=["text"] ) # ìš”ì•½ëœ ë‚´ìš©ë“¤ì„ ì·¨í•©í•˜ì—¬ ë‹¤ì‹œí•œë²ˆ ìš”ì•½í•˜ê¸° combine_prompt = PromptTemplate( template="""Combine all the youtube video transcripts provided within backticks ```{text}``` Provide a concise summary between 8 to 10 sentences. """, input_variables=["text"] ) chain = load_summarize_chain(llm, chain_type="map_reduce", verbose=False, map_prompt=prompt, combine_prompt=combine_prompt) result = chain.invoke(text) print(result['output_text'])
