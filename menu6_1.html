질문
다음에 제시된 조건에 따라 파이썬 코드를 생성해줘 # 조건 1. 현재 작업 폴더의 하위 폴더인 data 폴더의 sample1.pdf 파일이 작업 대상. 2. sample1.pdf 파일을 읽어 "일금"과 "부가세포함" 문자열 사이에 있는 숫자만를 추출. 3. 추출한 숫자를 정수로 변환하여 출력
답변
import PyPDF2 import re # PDF 파일 경로 설정 file_path = 'data/sample1.pdf' # PDF 파일 열기 with open(file_path, 'rb') as file: reader = PyPDF2.PdfReader(file) # PDF 내 모든 페이지의 텍스트 읽기 text = "" for page in reader.pages: text += page.extract_text() # '일금'과 '부가세포함' 사이의 숫자 찾기 pattern = '일금(.*?)부가세포함' match = re.search(pattern, text) if match: # 찾은 문자열에서 숫자만 추출 number_string = re.findall(r'\d+', match.group(1)) if number_string: number = int(''.join(number_string)) print("추출한 숫자:", number) else: print("숫자를 찾을 수 없습니다.") else: print("패턴을 찾을 수 없습니다.")
